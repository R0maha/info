Events

addEventListener(event, function, useCapture); removeEventListener("mousemove", myFunction); block.addEventListener('load', function, false);

принимаем объект события как аргумент с названием "e", название может быть любым однако наиболее распространены "e" или "event"

button.addEventListener('click', function(e) {
  console.log(e);
};

сработают все обработчики события "click" на элементе button, однако родительские элементы даже не узнают о том, что на button нажимали

button.addEventListener('click', function(e) {
  e.stopPropagation();
};

теперь браузер не будет переходить по ссылке сколько бы на нее не нажимал пользователь

button.addEventListener('click', function(e) {
  e.preventDefault();
});
